> name: generate-snake
> on:
>   schedule:
>     - cron: "0 */6 * * *"
>   workflow_dispatch:
>   push:
>     branches: [ main ]
> jobs:
>   generate:
>     runs-on: ubuntu-latest
>     steps:
>       - uses: Platane/snk@v3
>         with:
>           github_user_name: jamshaid-0206
>           outputs: |
>             dist/github-contribution-grid-snake.svg
>             dist/github-contribution-grid-snake-dark.svg?palette=github-dark
